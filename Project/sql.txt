drop table users;
drop table products;
drop table order_product;
drop table orders;
drop table shipments;
drop table payments;
drop table reviews;

create table if not exists users (
    id int primary key auto_increment,
    username varchar(20),
    password varchar(30),
    name varchar(30),
    email varchar(40),
    level int,
    address varchar(50),
    phone varchar(10)
);

create table if not exists products (
    id int primary key ,
    name varchar(50),
    category varchar(30),
    image text,
    price int ,
    quantity int
);

create table if not exists reviews(
    id int primary key auto_increment,
    user_id int,
    product_id int,
    review text
);

create table if not exists order_product(
    order_id int,
    product_id int,
    product_qty int
);

create table if not exists orders(
    id int primary key auto_increment,
    user_id int,
    phone varchar(10),
    address text,
    shipment_id int,
    payment_id int,
    date date,
    total int
);

create table if not exists shipments(
    id int primary key auto_increment,
    method text,
    fee int
);

create table if not exists payments(
    id int primary key auto_increment,
    method text
);
